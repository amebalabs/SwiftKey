# Developer tools and IDE shortcuts
- key: "d"
  icon: "terminal.fill"
  title: "Developer Tools"
  submenu:
    - key: "x"
      title: "Xcode"
      action: "launch:///Applications/Xcode.app"
    - key: "v"
      title: "Visual Studio Code"
      action: "launch:///Applications/Visual Studio Code.app"
    - key: "i"
      title: "IntelliJ IDEA"
      action: "launch:///Applications/IntelliJ IDEA.app"
    - key: "g"
      title: "Git Commands"
      submenu:
        - key: "p"
          title: "Pull"
          action: "shell://git pull"
          notify: true
        - key: "s"
          title: "Status"
          action: "shell://git status"
          notify: true
        - key: "c"
          title: "Commit"
          action: "shell://git commit -m \"Quick commit from SwiftKey\""
          notify: true
        - key: "b"
          title: "Checkout Branch"
          action: "shell://git branch | fzf | xargs git checkout"
          notify: true
    - key: "d"
      title: "Docker"
      submenu:
        - key: "p"
          title: "Docker ps"
          action: "shell://docker ps"
          notify: true
        - key: "i"
          title: "Docker images"
          action: "shell://docker images"
          notify: true
        - key: "c"
          title: "Docker compose up"
          action: "shell://docker-compose up -d"
          notify: true
