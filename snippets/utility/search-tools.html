# Search tools and quick lookups
- key: "?"
  icon: "magnifyingglass"
  title: "Search Tools"
  submenu:
    - key: "w"
      icon: "globe"
      title: "Web Search"
      submenu:
        - key: "g"
          title: "Google"
          action: "shell://read -p 'Search Google for: ' QUERY && open \"https://www.google.com/search?q=$QUERY\""
          notify: true
        - key: "d"
          title: "DuckDuckGo"
          action: "shell://read -p 'Search DuckDuckGo for: ' QUERY && open \"https://duckduckgo.com/?q=$QUERY\""
          notify: true
        - key: "y"
          title: "YouTube"
          action: "shell://read -p 'Search YouTube for: ' QUERY && open \"https://www.youtube.com/results?search_query=$QUERY\""
          notify: true
    - key: "l"
      icon: "folder"
      title: "Local Search"
      submenu:
        - key: "f"
          title: "Find File by Name"
          action: "shell://osascript -e 'tell application \"Finder\" to activate' -e 'tell application \"System Events\" to keystroke \"f\" using {command down}'"
          notify: true
        - key: "s"
          title: "Spotlight Search"
          action: "shell://osascript -e 'tell application \"System Events\" to keystroke space using {command down}'"
          notify: true
        - key: "t"
          title: "Terminal Find"
          action: "shell://read -p 'Find files matching: ' PATTERN && find . -name \"*$PATTERN*\" -type f | sort"
          notify: true
    - key: "d"
      icon: "book"
      title: "Documentation"
      submenu:
        - key: "a"
          title: "Apple Developer"
          action: "shell://read -p 'Search Apple Developer docs: ' QUERY && open \"https://developer.apple.com/search/?q=$QUERY\""
          notify: true
        - key: "m"
          title: "MDN Web Docs"
          action: "shell://read -p 'Search MDN Web Docs: ' QUERY && open \"https://developer.mozilla.org/en-US/search?q=$QUERY\""
          notify: true
        - key: "s"
          title: "Stack Overflow"
          action: "shell://read -p 'Search Stack Overflow: ' QUERY && open \"https://stackoverflow.com/search?q=$QUERY\""
          notify: true
