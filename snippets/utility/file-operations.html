# File operations and management tools
- key: "f"
  icon: "folder.fill"
  title: "File Operations"
  submenu:
    - key: "s"
      icon: "magnifyingglass"
      title: "Search"
      submenu:
        - key: "n"
          title: "Find by Name"
          action: "shell://read -p 'Enter filename pattern: ' PATTERN && find . -name "*$PATTERN*" -type f | head -20"
          notify: true
        - key: "t"
          title: "Find by Type"
          action: "shell://read -p 'Enter file extension: ' EXT && find . -name "*.$EXT" -type f | head -20"
          notify: true
        - key: "c"
          title: "Find Content in Files"
          action: "shell://read -p 'Enter text to search for: ' SEARCH && grep -r "$SEARCH" --include="*.txt" ."
          notify: true
    - key: "z"
      icon: "archivebox.fill"
      title: "Zip/Unzip"
      submenu:
        - key: "z"
          title: "Zip Current Directory"
          action: "shell://zip -r $(basename $(pwd)).zip ."
          notify: true
        - key: "u"
          title: "Unzip File"
          action: "shell://read -p 'Enter zip filename: ' ZIPFILE && unzip "$ZIPFILE""
          notify: true
    - key: "c"
      icon: "trash"
      title: "Cleanup"
      submenu:
        - key: "t"
          title: "Delete Temp Files"
          action: "shell://find . -name "*.tmp" -type f -delete"
          notify: true
        - key: "d"
          title: "Find Duplicate Files"
          action: "shell://find . -type f -exec md5 {} \; | sort | uniq -d -w32"
          notify: true
        - key: "e"
          title: "Empty Trash"
          action: "shell://osascript -e 'tell application \"Finder\" to empty trash'"
          notify: true
