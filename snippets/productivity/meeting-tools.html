# Meeting management and productivity tools
- key: "m"
  icon: "person.3.fill"
  title: "Meeting Tools"
  submenu:
    - key: "t"
      icon: "timer"
      title: "Timers"
      submenu:
        - key: "5"
          title: "5 Minute Timer"
          action: "shell://osascript -e 'display notification \"5 minute timer started\" with title \"Meeting Timer\"' && sleep 300 && osascript -e 'display notification \"Time is up!\" with title \"Meeting Timer\" sound name \"Glass\"'"
          notify: true
        - key: "1"
          title: "15 Minute Timer"
          action: "shell://osascript -e 'display notification \"15 minute timer started\" with title \"Meeting Timer\"' && sleep 900 && osascript -e 'display notification \"Time is up!\" with title \"Meeting Timer\" sound name \"Glass\"'"
          notify: true
        - key: "3"
          title: "30 Minute Timer"
          action: "shell://osascript -e 'display notification \"30 minute timer started\" with title \"Meeting Timer\"' && sleep 1800 && osascript -e 'display notification \"Time is up!\" with title \"Meeting Timer\" sound name \"Glass\"'"
          notify: true
    - key: "n"
      icon: "note.text"
      title: "Notes"
      submenu:
        - key: "c"
          title: "Create Meeting Notes"
          action: "shell://cat > ~/Desktop/meeting_notes_$(date +%Y%m%d).md << 'EOF'\n# Meeting Notes: $(date +%Y-%m-%d)\n\n## Attendees\n- \n\n## Agenda\n1. \n2. \n3. \n\n## Discussion\n\n\n## Action Items\n- [ ] \n- [ ] \n\n## Next Steps\n\nEOF\n&& open ~/Desktop/meeting_notes_$(date +%Y%m%d).md"
          notify: true
        - key: "a"
          title: "Add Action Item"
          action: "text://- [ ] Action: \n  Assigned to: \n  Due date: "
    - key: "z"
      icon: "video"
      title: "Zoom Controls"
      submenu:
        - key: "m"
          title: "Toggle Mute"
          action: "shell://osascript -e 'tell application \"System Events\" to keystroke \"a\" using {command down, shift down}'"
          notify: true
        - key: "v"
          title: "Toggle Video"
          action: "shell://osascript -e 'tell application \"System Events\" to keystroke \"v\" using {command down, shift down}'"
          notify: true
        - key: "s"
          title: "Start/Stop Screen Share"
          action: "shell://osascript -e 'tell application \"System Events\" to keystroke \"s\" using {command down, shift down}'"
          notify: true
