# Terminal commands and shortcuts
- key: "t"
  icon: "terminal"
  title: "Terminal Shortcuts"
  submenu:
    - key: "f"
      icon: "folder"
      title: "File Operations"
      submenu:
        - key: "l"
          title: "List Files (ls -la)"
          action: "shell://ls -la"
          notify: true
        - key: "d"
          title: "Disk Usage (du -h)"
          action: "shell://du -h -d 1 | sort -hr"
          notify: true
        - key: "f"
          title: "Find Large Files"
          action: "shell://find . -type f -size +100M -exec ls -lh {} \;"
          notify: true
    - key: "p"
      icon: "cpu"
      title: "Process Management"
      submenu:
        - key: "t"
          title: "Top Processes"
          action: "shell://ps aux | head -10"
          notify: true
        - key: "m"
          title: "Memory Usage"
          action: "shell://vm_stat"
          notify: true
        - key: "k"
          title: "Kill Process by Name"
          action: "shell://read -p 'Process name to kill: ' PROC && pkill "$PROC""
          notify: true
    - key: "n"
      icon: "network"
      title: "Network"
      submenu:
        - key: "i"
          title: "IP Information"
          action: "shell://ifconfig | grep inet"
          notify: true
        - key: "p"
          title: "Ping Google"
          action: "shell://ping -c 5 google.com"
          notify: true
        - key: "o"
          title: "Open Ports"
          action: "shell://lsof -i -P | grep LISTEN"
          notify: true
