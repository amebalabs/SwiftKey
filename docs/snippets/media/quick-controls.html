# Media playback controls for popular players
- key: "m"
  icon: "music.note"
  title: "Media Controls"
  submenu:
    - key: "p"
      icon: "play.fill"
      title: "Play/Pause"
      action: "shell://osascript -e 'tell application "Spotify" to playpause'"
    - key: "n"
      icon: "forward.fill"
      title: "Next Track"
      action: "shell://osascript -e 'tell application "Spotify" to next track'"
    - key: "b"
      icon: "backward.fill"
      title: "Previous Track"
      action: "shell://osascript -e 'tell application "Spotify" to previous track'"
    - key: "s"
      icon: "speaker.wave.2.fill"
      title: "Spotify"
      action: "launch:///Applications/Spotify.app"
    - key: "a"
      icon: "music.note"
      title: "Apple Music"
      action: "launch:///System/Applications/Music.app"
