# Quick app launcher by category
- key: "a"
  icon: "app.fill"
  title: "Applications"
  submenu:
    - key: "b"
      icon: "safari"
      title: "Browsers"
      submenu:
        - key: "s"
          title: "Safari"
          action: "launch:///Applications/Safari.app"
        - key: "c"
          title: "Chrome"
          action: "launch:///Applications/Google Chrome.app"
        - key: "f"
          title: "Firefox"
          action: "launch:///Applications/Firefox.app"
    - key: "p"
      icon: "doc.text"
      title: "Productivity"
      submenu:
        - key: "p"
          title: "Preview"
          action: "launch:///System/Applications/Preview.app"
        - key: "n"
          title: "Notes"
          action: "launch:///System/Applications/Notes.app"
        - key: "c"
          title: "Calculator"
          action: "launch:///System/Applications/Calculator.app"
