# Text transformation and formatting tools
- key: "x"
  icon: "text.format"
  title: "Text Tools"
  submenu:
    - key: "c"
      icon: "arrow.left.arrow.right"
      title: "Case Conversion"
      submenu:
        - key: "u"
          title: "UPPERCASE"
          action: "shell://pbpaste | tr '[:lower:]' '[:upper:]' | pbcopy && osascript -e 'display notification \"Text converted to uppercase\" with title \"Text Tools\"'"
          notify: true
        - key: "l"
          title: "lowercase"
          action: "shell://pbpaste | tr '[:upper:]' '[:lower:]' | pbcopy && osascript -e 'display notification \"Text converted to lowercase\" with title \"Text Tools\"'"
          notify: true
        - key: "t"
          title: "Title Case"
          action: "shell://pbpaste | python3 -c 'import sys; print(\" \".join(w.capitalize() for w in sys.stdin.read().split()))' | pbcopy && osascript -e 'display notification \"Text converted to title case\" with title \"Text Tools\"'"
          notify: true
    - key: "e"
      icon: "lock"
      title: "Encode/Decode"
      submenu:
        - key: "u"
          title: "URL Encode"
          action: "shell://pbpaste | python3 -c 'import sys, urllib.parse; print(urllib.parse.quote(sys.stdin.read().strip()))' | pbcopy && osascript -e 'display notification \"Text URL encoded\" with title \"Text Tools\"'"
          notify: true
        - key: "d"
          title: "URL Decode"
          action: "shell://pbpaste | python3 -c 'import sys, urllib.parse; print(urllib.parse.unquote(sys.stdin.read().strip()))' | pbcopy && osascript -e 'display notification \"Text URL decoded\" with title \"Text Tools\"'"
          notify: true
        - key: "b"
          title: "Base64 Encode"
          action: "shell://pbpaste | base64 | pbcopy && osascript -e 'display notification \"Text Base64 encoded\" with title \"Text Tools\"'"
          notify: true
        - key: "6"
          title: "Base64 Decode"
          action: "shell://pbpaste | base64 --decode | pbcopy && osascript -e 'display notification \"Text Base64 decoded\" with title \"Text Tools\"'"
          notify: true
    - key: "f"
      icon: "wand.and.stars"
      title: "Format"
      submenu:
        - key: "j"
          title: "Format JSON"
          action: "shell://pbpaste | python3 -m json.tool | pbcopy && osascript -e 'display notification \"JSON formatted\" with title \"Text Tools\"'"
          notify: true
        - key: "x"
          title: "Format XML"
          action: "shell://pbpaste | python3 -c 'import sys, xml.dom.minidom; print(xml.dom.minidom.parseString(sys.stdin.read()).toprettyxml())' | pbcopy && osascript -e 'display notification \"XML formatted\" with title \"Text Tools\"'"
          notify: true
        - key: "s"
          title: "Strip Whitespace"
          action: "shell://pbpaste | sed 's/^[ \t]*//;s/[ \t]*$//' | pbcopy && osascript -e 'display notification \"Whitespace stripped\" with title \"Text Tools\"'"
          notify: true
