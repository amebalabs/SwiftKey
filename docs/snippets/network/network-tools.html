# Network diagnostics and monitoring tools
- key: "n"
  icon: "network"
  title: "Network Tools"
  submenu:
    - key: "d"
      icon: "dial.min"
      title: "Diagnostics"
      submenu:
        - key: "p"
          title: "Ping Google DNS"
          action: "shell://ping -c 5 8.8.8.8"
          notify: true
        - key: "t"
          title: "Traceroute"
          action: "shell://traceroute google.com"
          notify: true
        - key: "f"
          title: "Flush DNS Cache"
          action: "shell://sudo dscacheutil -flushcache; sudo killall -HUP mDNSResponder"
          notify: true
    - key: "i"
      icon: "info.circle"
      title: "Network Info"
      submenu:
        - key: "i"
          title: "IP Address (Local)"
          action: "shell://ipconfig getifaddr en0"
          notify: true
        - key: "e"
          title: "External IP"
          action: "shell://curl -s https://api.ipify.org"
          notify: true
        - key: "d"
          title: "DNS Servers"
          action: "shell://scutil --dns | grep 'nameserver\[[0-9]*\]'"
          notify: true
        - key: "r"
          title: "Routing Table"
          action: "shell://netstat -nr"
          notify: true
    - key: "s"
      icon: "speedometer"
      title: "Speed Test"
      submenu:
        - key: "f"
          title: "Fast.com"
          action: "open://https://fast.com"
        - key: "o"
          title: "Ookla Speedtest"
          action: "open://https://www.speedtest.net"
        - key: "c"
          title: "Terminal Speed Test"
          action: "shell://curl -s https://raw.githubusercontent.com/sivel/speedtest-cli/master/speedtest.py | python -"
          notify: true
