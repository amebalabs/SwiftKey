# Window management and arrangement shortcuts
- key: "w"
  icon: "rectangle.3.group"
  title: "Window Management"
  submenu:
    - key: "h"
      icon: "rectangle.lefthalf"
      title: "Left Half"
      action: "shell://osascript -e 'tell application \"System Events\" to set frontApp to name of first application process whose frontmost is true' -e 'tell application \"System Events\" to tell process frontApp to set position of front window to {0, 22}' -e 'tell application \"System Events\" to tell process frontApp to set size of front window to {(get screen size's item 1)/2, (get screen size's item 2)-22}'"
      notify: true
    - key: "l"
      icon: "rectangle.righthalf"
      title: "Right Half"
      action: "shell://osascript -e 'tell application \"System Events\" to set frontApp to name of first application process whose frontmost is true' -e 'tell application \"System Events\" to tell process frontApp to set position of front window to {(get screen size's item 1)/2, 22}' -e 'tell application \"System Events\" to tell process frontApp to set size of front window to {(get screen size's item 1)/2, (get screen size's item 2)-22}'"
      notify: true
    - key: "f"
      icon: "rectangle.fill"
      title: "Full Screen"
      action: "shell://osascript -e 'tell application \"System Events\" to set frontApp to name of first application process whose frontmost is true' -e 'tell application \"System Events\" to tell process frontApp to set position of front window to {0, 22}' -e 'tell application \"System Events\" to tell process frontApp to set size of front window to {(get screen size's item 1), (get screen size's item 2)-22}'"
      notify: true
    - key: "q"
      icon: "rectangle.split.4x4"
      title: "Quarters"
      submenu:
        - key: "1"
          title: "Top Left"
          action: "shell://osascript -e 'tell application \"System Events\" to set frontApp to name of first application process whose frontmost is true' -e 'tell application \"System Events\" to tell process frontApp to set position of front window to {0, 22}' -e 'tell application \"System Events\" to tell process frontApp to set size of front window to {(get screen size's item 1)/2, (get screen size's item 2)/2-11}'"
          notify: true
        - key: "2"
          title: "Top Right"
          action: "shell://osascript -e 'tell application \"System Events\" to set frontApp to name of first application process whose frontmost is true' -e 'tell application \"System Events\" to tell process frontApp to set position of front window to {(get screen size's item 1)/2, 22}' -e 'tell application \"System Events\" to tell process frontApp to set size of front window to {(get screen size's item 1)/2, (get screen size's item 2)/2-11}'"
          notify: true
        - key: "3"
          title: "Bottom Left"
          action: "shell://osascript -e 'tell application \"System Events\" to set frontApp to name of first application process whose frontmost is true' -e 'tell application \"System Events\" to tell process frontApp to set position of front window to {0, (get screen size's item 2)/2+11}' -e 'tell application \"System Events\" to tell process frontApp to set size of front window to {(get screen size's item 1)/2, (get screen size's item 2)/2-11}'"
          notify: true
        - key: "4"
          title: "Bottom Right"
          action: "shell://osascript -e 'tell application \"System Events\" to set frontApp to name of first application process whose frontmost is true' -e 'tell application \"System Events\" to tell process frontApp to set position of front window to {(get screen size's item 1)/2, (get screen size's item 2)/2+11}' -e 'tell application \"System Events\" to tell process frontApp to set size of front window to {(get screen size's item 1)/2, (get screen size's item 2)/2-11}'"
          notify: true
    - key: "c"
      icon: "move.3d"
      title: "Center"
      action: "shell://osascript -e 'tell application \"System Events\" to set frontApp to name of first application process whose frontmost is true' -e 'tell application \"System Events\" to tell process frontApp to set position of front window to {(get screen size's item 1)/2-400, (get screen size's item 2)/2-300}' -e 'tell application \"System Events\" to tell process frontApp to set size of front window to {800, 600}'"
      notify: true
