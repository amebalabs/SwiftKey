# Clipboard history and quick text snippets
- key: "c"
  icon: "doc.on.clipboard"
  title: "Clipboard Manager"
  submenu:
    - key: "h"
      icon: "clock"
      title: "Clipboard History"
      submenu:
        - key: "1"
          title: "Copy to History Slot 1"
          action: "shell://osascript -e 'tell application \"System Events\" to set the clipboard to (the clipboard)' && defaults write com.swiftkey.app clipboard.slot1 -string "$(pbpaste)""
          notify: true
        - key: "2"
          title: "Copy to History Slot 2"
          action: "shell://osascript -e 'tell application \"System Events\" to set the clipboard to (the clipboard)' && defaults write com.swiftkey.app clipboard.slot2 -string "$(pbpaste)""
          notify: true
        - key: "3"
          title: "Copy to History Slot 3"
          action: "shell://osascript -e 'tell application \"System Events\" to set the clipboard to (the clipboard)' && defaults write com.swiftkey.app clipboard.slot3 -string "$(pbpaste)""
          notify: true
    - key: "p"
      icon: "arrow.down.doc"
      title: "Paste From History"
      submenu:
        - key: "1"
          title: "Paste From Slot 1"
          action: "shell://defaults read com.swiftkey.app clipboard.slot1 2>/dev/null | pbcopy"
          notify: true
        - key: "2"
          title: "Paste From Slot 2"
          action: "shell://defaults read com.swiftkey.app clipboard.slot2 2>/dev/null | pbcopy"
          notify: true
        - key: "3"
          title: "Paste From Slot 3"
          action: "shell://defaults read com.swiftkey.app clipboard.slot3 2>/dev/null | pbcopy"
          notify: true
    - key: "t"
      icon: "text.badge.plus"
      title: "Text Snippets"
      submenu:
        - key: "e"
          title: "Email Signature"
          action: "text://Best regards,\n[Your Name]\n[Your Position]\n[Your Contact Info]"
        - key: "m"
          title: "Meeting Template"
          action: "text://Meeting: [Title]\nDate: [Date]\nAttendees: [Attendees]\nAgenda:\n1. \n2. \n3. "
        - key: "t"
          title: "Thank You"
          action: "text://Thank you for your assistance. I really appreciate your help with this matter."
