# macOS system utilities and controls
- key: "s"
  icon: "gearshape.fill"
  title: "System"
  submenu:
    - key: "d"
      icon: "display"
      title: "Display"
      submenu:
        - key: "n"
          title: "Night Shift Toggle"
          action: "shell://osascript -e 'tell application \"System Events\" to tell appearance preferences to set dark mode to not dark mode'"
          notify: true
        - key: "b"
          title: "Brightness Up"
          action: "shell://osascript -e 'tell application \"System Events\" to key code 144'"
          notify: true
        - key: "d"
          title: "Brightness Down"
          action: "shell://osascript -e 'tell application \"System Events\" to key code 145'"
          notify: true
    - key: "a"
      icon: "speaker.wave.3.fill"
      title: "Audio"
      submenu:
        - key: "u"
          title: "Volume Up"
          action: "shell://osascript -e 'set volume output volume (output volume of (get volume settings) + 10)'"
          notify: true
        - key: "d"
          title: "Volume Down"
          action: "shell://osascript -e 'set volume output volume (output volume of (get volume settings) - 10)'"
          notify: true
        - key: "m"
          title: "Mute Toggle"
          action: "shell://osascript -e 'set volume with output muted not (output muted of (get volume settings))'"
          notify: true
    - key: "p"
      icon: "bolt.fill"
      title: "Power"
      submenu:
        - key: "s"
          title: "Sleep"
          action: "shell://pmset sleepnow"
          notify: true
        - key: "r"
          title: "Restart"
          action: "shell://osascript -e 'tell app \"System Events\" to restart'"
          notify: true
        - key: "o"
          title: "Shut Down"
          action: "shell://osascript -e 'tell app \"System Events\" to shut down'"
          notify: true
