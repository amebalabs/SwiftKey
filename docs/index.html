Configuration


- key: "a"
  icon: "star.fill"
  title: "Launch Calculator"
  action: "launch://Calculator"
  submenu:
    - key: "b"
      icon: "safari"
      title: "Open Website"
      action: "open://https://example.com"

- key: "s"
  icon: "terminal"
  title: "Run Command"
  action: "shell://echo 'Hello'"
  sticky: true
