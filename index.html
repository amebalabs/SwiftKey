Configuration


# SwiftKey Configuration Example
- key: "m"
  icon: "message.fill"
  title: "Messaging"
  submenu:
    - key: "s"
      title: "Slack"
      action: "launch:///Applications/Slack.app"
    - key: "z"
      title: "Zoom"
      action: "launch:///Applications/zoom.us.app"
    - key: "m"
      title: "Messages"
      action: "launch:///System/Applications/Messages.app/"
- key: "s"
  icon: "terminal"
  title: "Run Command"
  action: "shell://'say beep'"
  sticky: true
